```go
// internal/challenge/types.go
package challenge

// Challenge represents a single learning challenge.
type Challenge struct {
	ID          string       `yaml:"id"`
	Title       string       `yaml:"title"`
	Difficulty  int          `yaml:"difficulty"`
	Category    string       `yaml:"category"`
	Subcategory string       `yaml:"subcategory"`
	Tags        []string     `yaml:"tags"`
	Description string       `yaml:"description"`
	Hints       []Hint       `yaml:"hints,omitempty"`
	Verify      []VerifyRule `yaml:"verify"`
	SetupFiles  []SetupFile  `yaml:"setup_files,omitempty"`
}

// Hint is a progressive hint that can be unlocked level by level.
type Hint struct {
	Level int    `yaml:"level"`
	Text  string `yaml:"text"`
}

// VerifyRule defines one verification check for a challenge.
type VerifyRule struct {
	Type    string `yaml:"type"`
	Path    string `yaml:"path,omitempty"`
	Expect  string `yaml:"expect,omitempty"`
	Command string `yaml:"command,omitempty"`
}

// SetupFile defines a file to be created before a Vim challenge.
type SetupFile struct {
	Path    string `yaml:"path"`
	Content string `yaml:"content"`
}
```

```go
// internal/challenge/loader.go
package challenge

import (
	"fmt"
	"os"
	"path/filepath"
	"sort"

	"gopkg.in/yaml.v3"
)

// LoadChallenge loads a single challenge from a directory containing challenge.yaml.
func LoadChallenge(dir string) (*Challenge, error) {
	data, err := os.ReadFile(filepath.Join(dir, "challenge.yaml"))
	if err != nil {
		return nil, fmt.Errorf("read challenge.yaml: %w", err)
	}
	var c Challenge
	if err := yaml.Unmarshal(data, &c); err != nil {
		return nil, fmt.Errorf("parse challenge.yaml: %w", err)
	}
	c.Dir = dir
	return &c, nil
}

// LoadAll loads all challenges from the root challenges directory.
// It expects: root/<category>/<challenge-name>/challenge.yaml
func LoadAll(root string) ([]*Challenge, error) {
	var challenges []*Challenge

	categories, err := os.ReadDir(root)
	if err != nil {
		return nil, fmt.Errorf("read challenges root: %w", err)
	}

	for _, cat := range categories {
		if !cat.IsDir() {
			continue
		}
		catPath := filepath.Join(root, cat.Name())
		entries, err := os.ReadDir(catPath)
		if err != nil {
			continue
		}
		for _, entry := range entries {
			if !entry.IsDir() {
				continue
			}
			chDir := filepath.Join(catPath, entry.Name())
			c, err := LoadChallenge(chDir)
			if err != nil {
				continue
			}
			challenges = append(challenges, c)
		}
	}

	sort.Slice(challenges, func(i, j int) bool {
		if challenges[i].Category != challenges[j].Category {
			return challenges[i].Category < challenges[j].Category
		}
		return challenges[i].ID < challenges[j].ID
	})

	return challenges, nil
}

// LoadAllByCategory loads all challenges grouped by category.
func LoadAllByCategory(root string) (map[string][]*Challenge, error) {
	all, err := LoadAll(root)
	if err != nil {
		return nil, err
	}
	result := make(map[string][]*Challenge)
	for _, c := range all {
		result[c.Category] = append(result[c.Category], c)
	}
	return result, nil
}
```

```go
// internal/verify/verifier.go
package verify

import "github.com/sd3/linuxlab/internal/challenge"

// Result is the outcome of a single verification check.
type Result struct {
	Passed  bool
	Message string
}

// Verifier checks a single VerifyRule and returns a Result.
type Verifier interface {
	Verify(rule challenge.VerifyRule) Result
}
```

```go
// internal/verify/file.go
package verify

import (
	"fmt"
	"os"
	"strings"

	"github.com/sd3/linuxlab/internal/challenge"
)

// FileContentVerifier checks that a file's content matches the expected string.
type FileContentVerifier struct{}

func (v *FileContentVerifier) Verify(rule challenge.VerifyRule) Result {
	data, err := os.ReadFile(rule.Path)
	if err != nil {
		return Result{Passed: false, Message: fmt.Sprintf("无法读取文件 %s: %v", rule.Path, err)}
	}

	actual := strings.TrimSpace(string(data))
	expect := strings.TrimSpace(rule.Expect)

	if actual != expect {
		return Result{
			Passed:  false,
			Message: fmt.Sprintf("文件内容不匹配\n期望:\n%s\n实际:\n%s", expect, actual),
		}
	}

	return Result{Passed: true, Message: "文件内容匹配"}
}

// FileExistsVerifier checks that a file or directory exists.
type FileExistsVerifier struct{}

func (v *FileExistsVerifier) Verify(rule challenge.VerifyRule) Result {
	_, err := os.Stat(rule.Path)
	if err != nil {
		return Result{Passed: false, Message: fmt.Sprintf("路径不存在: %s", rule.Path)}
	}
	return Result{Passed: true, Message: fmt.Sprintf("路径存在: %s", rule.Path)}
}
```

```go
	output, err := cmd.Output()
	if err != nil {
		return Result{Passed: false, Message: fmt.Sprintf("命令执行失败: %v", err)}
	}

	if actual != expect {
		return Result{
			Passed:  false,
			Message: fmt.Sprintf("输出不匹配\n期望:\n%s\n实际:\n%s", expect, actual),
		}
	}

	return Result{Passed: true, Message: "命令输出匹配"}
}

// ExitCodeVerifier runs a command and checks its exit code.
type ExitCodeVerifier struct{}

func (v *ExitCodeVerifier) Verify(rule challenge.VerifyRule) Result {
	expectedCode, err := strconv.Atoi(strings.TrimSpace(rule.Expect))
	if err != nil {
		return Result{Passed: false, Message: fmt.Sprintf("无效的期望退出码: %s", rule.Expect)}
	}

	cmd := exec.Command("sh", "-c", rule.Command)
	err = cmd.Run()

	actualCode := 0
	if err != nil {
		if exitErr, ok := err.(*exec.ExitError); ok {
			actualCode = exitErr.ExitCode()
		} else {
			return Result{Passed: false, Message: fmt.Sprintf("命令执行失败: %v", err)}
		}
	}

	if actualCode != expectedCode {
		return Result{
			Passed:  false,
			Message: fmt.Sprintf("退出码不匹配: 期望 %d, 实际 %d", expectedCode, actualCode),
		}
	}

	return Result{Passed: true, Message: fmt.Sprintf("退出码匹配: %d", expectedCode)}
```

```go
// internal/verify/verifier.go
package verify

import (
	"fmt"

	"github.com/sd3/linuxlab/internal/challenge"
)

// Result is the outcome of a single verification check.
type Result struct {
	Passed  bool
	Message string
}

// Verifier checks a single VerifyRule and returns a Result.
type Verifier interface {
	Verify(rule challenge.VerifyRule) Result
}

var registry = map[string]Verifier{
	"file_content":   &FileContentVerifier{},
	"file_exists":    &FileExistsVerifier{},
	"command_output": &CommandOutputVerifier{},
	"exit_code":      &ExitCodeVerifier{},
}

// RunAll executes all verification rules and returns results.
func RunAll(rules []challenge.VerifyRule) []Result {
	results := make([]Result, len(rules))
	for i, rule := range rules {
		v, ok := registry[rule.Type]
		if !ok {
			results[i] = Result{Passed: false, Message: fmt.Sprintf("未知验证类型: %s", rule.Type)}
			continue
		}
		results[i] = v.Verify(rule)
	}
	return results
}

// AllPassed returns true if every result passed.
func AllPassed(results []Result) bool {
	for _, r := range results {
		if !r.Passed {
			return false
		}
	}
	return true
}
```

```go
// internal/progress/store.go
package progress

import (
	"encoding/json"
	"os"
	"time"
)

// ProgressData is the top-level structure persisted to JSON.
type ProgressData struct {
	Skills     map[string]*SkillEntry     `json:"skills"`
	Challenges map[string]*ChallengeEntry `json:"challenges"`
}

// SkillEntry tracks stats for a subcategory (e.g. "linux-basics.file-operations").
type SkillEntry struct {
	Total  int     `json:"total"`
	Passed int     `json:"passed"`
	Score  float64 `json:"score"`
}

// ChallengeEntry tracks a single challenge's attempt history.
type ChallengeEntry struct {
	Status      string `json:"status"`
	Attempts    int    `json:"attempts"`
	HintsUsed   int    `json:"hints_used"`
	LastAttempt string `json:"last_attempt"`
}

// Store manages progress persistence.
type Store struct {
	Data ProgressData
	path string
}

// NewStore loads progress from path, or creates empty progress if the file doesn't exist.
func NewStore(path string) (*Store, error) {
	s := &Store{
		path: path,
		Data: ProgressData{
			Skills:     make(map[string]*SkillEntry),
			Challenges: make(map[string]*ChallengeEntry),
		},
	}

	data, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return s, nil
		}
		return nil, err
	}

	if err := json.Unmarshal(data, &s.Data); err != nil {
		return nil, err
	}

	if s.Data.Skills == nil {
		s.Data.Skills = make(map[string]*SkillEntry)
	}
	if s.Data.Challenges == nil {
		s.Data.Challenges = make(map[string]*ChallengeEntry)
	}

	return s, nil
}

// RecordAttempt records the result of one challenge attempt.
func (s *Store) RecordAttempt(challengeID, category, subcategory string, passed bool, hintsUsed int) {
	entry, ok := s.Data.Challenges[challengeID]
	if !ok {
		entry = &ChallengeEntry{}
		s.Data.Challenges[challengeID] = entry
	}

	wasPassed := entry.Status == "passed"
	entry.Attempts++
	entry.LastAttempt = time.Now().Format("2006-01-02")
	if hintsUsed > entry.HintsUsed {
		entry.HintsUsed = hintsUsed
	}
	if passed {
		entry.Status = "passed"
	} else if entry.Status == "" {
		entry.Status = "failed"
	}

	// Update skill entry
	skillKey := category + "." + subcategory
	skill, ok := s.Data.Skills[skillKey]
	if !ok {
		skill = &SkillEntry{}
		s.Data.Skills[skillKey] = skill
	}

	if !wasPassed && !ok {
		skill.Total++
	} else if !ok {
		skill.Total++
	} else if !wasPassed && entry.Attempts == 1 {
		skill.Total++
	}

	if passed && !wasPassed {
		skill.Passed++
	}

	if skill.Total > 0 {
		skill.Score = float64(skill.Passed) / float64(skill.Total)
	}
}

// Save persists progress data to disk.
func (s *Store) Save() error {
	data, err := json.MarshalIndent(s.Data, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(s.path, data, 0o644)
}
```

```go
// internal/progress/skillmap.go
package progress

import (
	"math"
	"sort"
	"strings"

	"github.com/sd3/linuxlab/internal/challenge"
)

// SkillMap is the computed skill overview for display.
type SkillMap struct {
	Categories   []CategoryScore
	TotalPassed  int
	TotalCount   int
	OverallScore float64
}

// CategoryScore aggregates subcategory scores for one module.
type CategoryScore struct {
	Name          string
	Score         float64
	Subcategories []SubcategoryScore
}

// SubcategoryScore holds stats for one subcategory.
type SubcategoryScore struct {
	Name   string
	Total  int
	Passed int
	Score  float64
}

// BuildSkillMap computes the skill map from the store.
func BuildSkillMap(store *Store) *SkillMap {
	catMap := make(map[string][]SubcategoryScore)

	totalPassed := 0
	totalCount := 0

	for key, skill := range store.Data.Skills {
		parts := strings.SplitN(key, ".", 2)
		if len(parts) != 2 {
			continue
		}
		cat, sub := parts[0], parts[1]
		catMap[cat] = append(catMap[cat], SubcategoryScore{
			Name:   sub,
			Total:  skill.Total,
			Passed: skill.Passed,
			Score:  skill.Score,
		})
		totalPassed += skill.Passed
		totalCount += skill.Total
	}

	var categories []CategoryScore
	for name, subs := range catMap {
		sort.Slice(subs, func(i, j int) bool { return subs[i].Name < subs[j].Name })
		avg := 0.0
		for _, s := range subs {
			avg += s.Score
		}
		if len(subs) > 0 {
			avg /= float64(len(subs))
		}
		categories = append(categories, CategoryScore{
			Name:          name,
			Score:         avg,
			Subcategories: subs,
		})
	}
	sort.Slice(categories, func(i, j int) bool { return categories[i].Name < categories[j].Name })

	overall := 0.0
	if totalCount > 0 {
		overall = float64(totalPassed) / float64(totalCount)
	}

	return &SkillMap{
		Categories:   categories,
		TotalPassed:  totalPassed,
		TotalCount:   totalCount,
		OverallScore: overall,
	}
}

// RecommendWeakest returns a challenge from the weakest subcategory that hasn't been passed.
func RecommendWeakest(store *Store, challenges []*challenge.Challenge) *challenge.Challenge {
	// Find the skill with the lowest score
	var weakestKey string
	weakestScore := math.MaxFloat64

	for key, skill := range store.Data.Skills {
		if skill.Score < weakestScore {
			weakestScore = skill.Score
			weakestKey = key
		}
	}

	if weakestKey == "" {
		return nil
	}

	parts := strings.SplitN(weakestKey, ".", 2)
	if len(parts) != 2 {
		return nil
	}
	cat, sub := parts[0], parts[1]

	// Find an unpassed challenge in that subcategory
	for _, c := range challenges {
		if c.Category == cat && c.Subcategory == sub {
			entry := store.Data.Challenges[c.ID]
			if entry == nil || entry.Status != "passed" {
				return c
			}
		}
	}

	return nil
}

// ScoreWithHints computes the score penalty for using hints.
// Each hint multiplies score by 0.8.
func ScoreWithHints(hintsUsed int) float64 {
	return math.Pow(0.8, float64(hintsUsed))
}
```

```go
// internal/sandbox/docker.go
package sandbox

import (
	"bytes"
	"context"
	"fmt"
	"io"

	"github.com/docker/docker/api/types"
	"github.com/docker/docker/api/types/container"
	"github.com/docker/docker/client"
	"github.com/docker/docker/pkg/stdcopy"
)

const defaultImage = "ubuntu:22.04"

// DockerSandbox manages a Docker container for challenge execution.
type DockerSandbox struct {
	cli         *client.Client
	containerID string
}

// NewDockerSandbox creates and starts a new container.
func NewDockerSandbox(ctx context.Context, image string) (*DockerSandbox, error) {
	if image == "" {
		image = defaultImage
	}

	cli, err := client.NewClientWithOpts(client.FromEnv, client.WithAPIVersionNegotiation())
	if err != nil {
		return nil, fmt.Errorf("docker client: %w", err)
	}

	resp, err := cli.ContainerCreate(ctx,
		&container.Config{
			Image: image,
			Cmd:   []string{"sleep", "3600"},
			Tty:   false,
		},
		&container.HostConfig{
			Resources: container.Resources{
				Memory:   512 * 1024 * 1024,
				NanoCPUs: 1_000_000_000,
			},
		},
		nil, nil, "",
	)
	if err != nil {
		cli.Close()
		return nil, fmt.Errorf("container create: %w", err)
	}

	if err := cli.ContainerStart(ctx, resp.ID, container.StartOptions{}); err != nil {
		cli.ContainerRemove(ctx, resp.ID, container.RemoveOptions{Force: true})
		cli.Close()
		return nil, fmt.Errorf("container start: %w", err)
	}

	return &DockerSandbox{cli: cli, containerID: resp.ID}, nil
}

// ContainerID returns the Docker container ID.
func (s *DockerSandbox) ContainerID() string {
	return s.containerID
}

// Exec runs a command inside the container and returns stdout, exit code.
func (s *DockerSandbox) Exec(ctx context.Context, command string) (string, int, error) {
	execCfg := container.ExecOptions{
		Cmd:          []string{"sh", "-c", command},
		AttachStdout: true,
		AttachStderr: true,
	}

	execResp, err := s.cli.ContainerExecCreate(ctx, s.containerID, execCfg)
	if err != nil {
		return "", -1, fmt.Errorf("exec create: %w", err)
	}

	attachResp, err := s.cli.ContainerExecAttach(ctx, execResp.ID, container.ExecStartOptions{})
	if err != nil {
		return "", -1, fmt.Errorf("exec attach: %w", err)
	}
	defer attachResp.Close()

	var stdout, stderr bytes.Buffer
	_, err = stdcopy.StdCopy(&stdout, &stderr, attachResp.Reader)
	if err != nil {
		return "", -1, fmt.Errorf("read output: %w", err)
	}

	inspectResp, err := s.cli.ContainerExecInspect(ctx, execResp.ID)
	if err != nil {
		return stdout.String(), -1, fmt.Errorf("exec inspect: %w", err)
	}

	return stdout.String(), inspectResp.ExitCode, nil
}

// Destroy stops and removes the container.
func (s *DockerSandbox) Destroy(ctx context.Context) error {
	err := s.cli.ContainerRemove(ctx, s.containerID, container.RemoveOptions{Force: true})
	s.cli.Close()
	return err
}

// CopyFileToContainer copies content into the container at the given path.
func (s *DockerSandbox) CopyFileToContainer(ctx context.Context, content, destPath string) error {
	// Use exec to write file — simpler than the archive-based copy API
	cmd := fmt.Sprintf("mkdir -p $(dirname %s) && cat > %s", destPath, destPath)
	execCfg := container.ExecOptions{
		Cmd:          []string{"sh", "-c", cmd},
		AttachStdin:  true,
		AttachStdout: true,
		AttachStderr: true,
	}

	execResp, err := s.cli.ContainerExecCreate(ctx, s.containerID, execCfg)
	if err != nil {
		return err
	}

	attachResp, err := s.cli.ContainerExecAttach(ctx, execResp.ID, container.ExecStartOptions{})
	if err != nil {
		return err
	}
	defer attachResp.Close()

	_, err = io.WriteString(attachResp.Conn, content)
	if err != nil {
		return err
	}
	attachResp.CloseWrite()

	// Drain output
	io.Copy(io.Discard, attachResp.Reader)
	return nil
}

// InteractiveShellArgs returns the docker exec command args for launching
// an interactive shell inside this container. The caller (TUI) will use
// tea.ExecProcess to hand control to the user.
func (s *DockerSandbox) InteractiveShellArgs() []string {
	return []string{"docker", "exec", "-it", s.containerID, "/bin/bash"}
}
```

```go
package sandbox

import (
	"os"
	"path/filepath"

	"github.com/sd3/linuxlab/internal/challenge"
	"github.com/sd3/linuxlab/internal/verify"
)

// VimRunner manages Vim challenge file preparation and verification.
type VimRunner struct {
	WorkDir string
}

// PrepareFiles writes setup files to the work directory and returns their absolute paths.
func (r *VimRunner) PrepareFiles(files []challenge.SetupFile) ([]string, error) {
	var paths []string
	for _, f := range files {
		fullPath := filepath.Join(r.WorkDir, f.Path)
		dir := filepath.Dir(fullPath)
		if err := os.MkdirAll(dir, 0o755); err != nil {
			return nil, err
		}
		if err := os.WriteFile(fullPath, []byte(f.Content), 0o644); err != nil {
			return nil, err
		}
		paths = append(paths, fullPath)
	}
	return paths, nil
}

// VimArgs returns the command and arguments to launch Vim for editing.
func (r *VimRunner) VimArgs(filePath string) []string {
	return []string{"vim", filePath}
}

// CheckResult runs the verification rules against the edited files.
func (r *VimRunner) CheckResult(rules []challenge.VerifyRule) bool {
	results := verify.RunAll(rules)
	return verify.AllPassed(results)
}
```

```go
	s := ""
	for i := 0; i < 5; i++ {
		if i < level {

		} else {
			s += empty.Render("☆")
		}
	}
	return s
}

// ProgressBar renders a text-based progress bar.
func ProgressBar(ratio float64, width int) string {
	filled := int(ratio * float64(width))
	if filled > width {
		filled = width
	}
	bar := ""
	for i := 0; i < width; i++ {
		if i < filled {
			bar += ProgressFull
		} else {
			bar += ProgressEmpty
		}
	}
	return bar
}
```

```go
package tui

import (
	"fmt"
	"strings"

	tea "github.com/charmbracelet/bubbletea"
)

// MenuChoiceMsg is sent when the user selects a menu item.
type MenuChoiceMsg struct {
	Choice string
}

type menuItem struct {
	label string
	key   string
}

// MenuModel is the main menu screen.
type MenuModel struct {
	items  []menuItem
	cursor int
}

// NewMenuModel creates the main menu.
func NewMenuModel() MenuModel {
	return MenuModel{
		items: []menuItem{
			{label: "开始练习", key: "practice"},
			{label: "能力图谱", key: "skillmap"},
			{label: "薄弱推荐", key: "recommend"},
			{label: "命令速查", key: "reference"},
		},
		cursor: 0,
	}
}

func (m MenuModel) Init() tea.Cmd {
	return nil
}

func (m MenuModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch {
		case msg.Type == tea.KeyUp || (msg.Type == tea.KeyRunes && string(msg.Runes) == "k"):
			m.cursor--
			if m.cursor < 0 {
				m.cursor = len(m.items) - 1
			}
		case msg.Type == tea.KeyDown || (msg.Type == tea.KeyRunes && string(msg.Runes) == "j"):
			m.cursor++
			if m.cursor >= len(m.items) {
				m.cursor = 0
			}
		case msg.Type == tea.KeyEnter:
			choice := m.items[m.cursor].key
			return m, func() tea.Msg { return MenuChoiceMsg{Choice: choice} }
		case msg.Type == tea.KeyRunes && string(msg.Runes) == "q":
			return m, tea.Quit
		}
	}
	return m, nil
}

func (m MenuModel) View() string {
	var b strings.Builder

	title := TitleStyle.Render("LinuxLab")
	b.WriteString("\n")

	var lines []string
	for i, item := range m.items {
		cursor := "  "
		style := DimStyle
		if i == m.cursor {
			cursor = SelectedStyle.Render("▸ ")
			style = SelectedStyle
		}
		lines = append(lines, fmt.Sprintf("%s%s", cursor, style.Render(item.label)))
	}

	lines = append(lines, "")
	lines = append(lines, HelpStyle.Render("↑↓/jk 选择  Enter 确认  q 退出"))

	content := strings.Join(lines, "\n")
	box := BoxStyle.Width(40).Render(fmt.Sprintf("%s\n\n%s", title, content))
	b.WriteString(box)
	b.WriteString("\n")

	return b.String()
}
```

```go
// ModulesModel is the module selection screen.
type ModulesModel struct {
	modules []moduleInfo
	cursor  int
}

var categoryLabels = map[string]string{
	"linux-basics":    "Linux 基础命令",
	"vim":             "Vim 操作",
	"shell-scripting": "Shell 脚本",
	"ops":             "运维实战",
	"containers":      "容器与部署",
}

// NewModulesModel creates the module selection screen.
func NewModulesModel(byCategory map[string][]*challenge.Challenge, store *progress.Store) ModulesModel {
	var modules []moduleInfo

	for cat, challenges := range byCategory {
		label, ok := categoryLabels[cat]
		if !ok {
			label = cat
		}
		passed := 0
		for _, c := range challenges {
			if entry, ok := store.Data.Challenges[c.ID]; ok && entry.Status == "passed" {
				passed++
			}
		}
		modules = append(modules, moduleInfo{
			category: cat,
			label:    label,
			total:    len(challenges),
			passed:   passed,
		})
	}

	sort.Slice(modules, func(i, j int) bool {
		return modules[i].category < modules[j].category
	})

	return ModulesModel{modules: modules}
}

func (m ModulesModel) Init() tea.Cmd { return nil }

func (m ModulesModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {

			m.cursor--
			if m.cursor < 0 {

			}

			m.cursor++
			if m.cursor >= len(m.modules) {
				m.cursor = 0
			}
		case msg.Type == tea.KeyEnter:
			cat := m.modules[m.cursor].category
			return m, func() tea.Msg { return ModuleSelectedMsg{Category: cat} }
		case msg.Type == tea.KeyEsc || (msg.Type == tea.KeyRunes && string(msg.Runes) == "q"):
```

```go
// internal/tui/app.go
package tui

import (
	"os/exec"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/sd3/linuxlab/internal/challenge"
	"github.com/sd3/linuxlab/internal/progress"
	"github.com/sd3/linuxlab/internal/verify"
)

type screenID int

const (
	screenMenu screenID = iota
	screenModules
	screenChallenges
	screenDetail
	screenSkillMap
	screenResult
)

// ChallengeResultMsg is sent after a challenge attempt completes.
type ChallengeResultMsg struct {
	Passed     bool
	Results    []verify.Result
	HintsUsed int
}

// AppModel is the root Bubbletea model that manages screen transitions.
type AppModel struct {
	screen         screenID
	menu           MenuModel
	modules        ModulesModel
	challenges     ChallengesModel
	detail         DetailModel
	skillmap       SkillMapModel
	byCategory     map[string][]*challenge.Challenge
	store          *progress.Store
	currentChallenge *challenge.Challenge
	lastResult     *ChallengeResultMsg
}

// NewAppModel creates the root app model.
func NewAppModel(byCategory map[string][]*challenge.Challenge, store *progress.Store) AppModel {
	return AppModel{
		screen:     screenMenu,
		menu:       NewMenuModel(),
		byCategory: byCategory,
		store:      store,
	}
}

func (m AppModel) Init() tea.Cmd {
	return nil
}

func (m AppModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {

	// ---- Screen navigation messages ----
	case MenuChoiceMsg:
		switch msg.Choice {
		case "practice":
			m.modules = NewModulesModel(m.byCategory, m.store)
			m.screen = screenModules
		case "skillmap":
			m.skillmap = NewSkillMapModel(m.store)
			m.screen = screenSkillMap
		case "recommend":
			// TODO P1: implement recommend screen
		case "reference":
			// TODO P1: implement reference screen
		}
		return m, nil

	case ModuleSelectedMsg:
		challenges := m.byCategory[msg.Category]
		m.challenges = NewChallengesModel(msg.Category, challenges, m.store)
		m.screen = screenChallenges
		return m, nil

	case ChallengeSelectedMsg:
		m.currentChallenge = msg.Challenge
		m.detail = NewDetailModel(msg.Challenge)
		m.screen = screenDetail
		return m, nil

	case LaunchChallengeMsg:
		return m, m.launchChallenge(msg)

	case ChallengeResultMsg:
		m.lastResult = &msg
		ch := m.currentChallenge
		m.store.RecordAttempt(ch.ID, ch.Category, ch.Subcategory, msg.Passed, msg.HintsUsed)
		m.store.Save()
		m.screen = screenResult
		return m, nil

	case GoBackMsg:
		switch m.screen {
		case screenModules:
			m.screen = screenMenu
		case screenChallenges:
			m.screen = screenModules
		case screenDetail:
			m.screen = screenChallenges
		case screenSkillMap:
			m.screen = screenMenu
		case screenResult:
			m.screen = screenChallenges
		}
		return m, nil
	}

	// ---- Delegate to active screen ----
	var cmd tea.Cmd
	switch m.screen {
	case screenMenu:
		var updated tea.Model
		updated, cmd = m.menu.Update(msg)
		m.menu = updated.(MenuModel)
	case screenModules:
		var updated tea.Model
		updated, cmd = m.modules.Update(msg)
		m.modules = updated.(ModulesModel)
	case screenChallenges:
		var updated tea.Model
		updated, cmd = m.challenges.Update(msg)
		m.challenges = updated.(ChallengesModel)
	case screenDetail:
		var updated tea.Model
		updated, cmd = m.detail.Update(msg)
		m.detail = updated.(DetailModel)
	case screenSkillMap:
		var updated tea.Model
		updated, cmd = m.skillmap.Update(msg)
		m.skillmap = updated.(SkillMapModel)
	case screenResult:
		if keyMsg, ok := msg.(tea.KeyMsg); ok {
			switch {
			case keyMsg.Type == tea.KeyEnter:
				return m, func() tea.Msg { return GoBackMsg{} }
			case keyMsg.Type == tea.KeyEsc || (keyMsg.Type == tea.KeyRunes && string(keyMsg.Runes) == "q"):
				return m, func() tea.Msg { return GoBackMsg{} }
			}
		}
	}
	return m, cmd
}

func (m AppModel) View() string {
	switch m.screen {
	case screenMenu:
		return m.menu.View()
	case screenModules:
		return m.modules.View()
	case screenChallenges:
		return m.challenges.View()
	case screenDetail:
		return m.detail.View()
	case screenSkillMap:
		return m.skillmap.View()
	case screenResult:
		return m.resultView()
	}
	return ""
}

func (m AppModel) resultView() string {
	if m.lastResult == nil {
		return ""
	}

	icon := PassedIcon
	msg := SelectedStyle.Render("通过！")
	if !m.lastResult.Passed {
		icon = FailedIcon
		msg = ErrorStyle.Render("未通过")
	}

	content := icon + " " + msg + "\n\n"
	for _, r := range m.lastResult.Results {
		rIcon := PassedIcon
		if !r.Passed {
			rIcon = FailedIcon
		}
		content += "  " + rIcon + " " + r.Message + "\n"
	}
	content += "\n" + HelpStyle.Render("Enter 返回列表  s 查看参考答案")

	return "\n" + BoxStyle.Width(55).Render(TitleStyle.Render("检测结果")+"\n\n"+content) + "\n"
}

func (m AppModel) launchChallenge(msg LaunchChallengeMsg) tea.Cmd {
	ch := msg.Challenge
	hintsUsed := msg.HintsUsed

	if ch.Category == "vim" {
		return m.launchVimChallenge(ch, hintsUsed)
	}

	return m.launchDockerChallenge(ch, hintsUsed)
}

func (m AppModel) launchVimChallenge(ch *challenge.Challenge, hintsUsed int) tea.Cmd {
	if len(ch.SetupFiles) == 0 {
		return func() tea.Msg {
			return ChallengeResultMsg{Passed: false, Results: []verify.Result{{Message: "题目缺少 setup_files"}}}
		}
	}

	// Write setup files to temp dir, launch vim, then verify
	return func() tea.Msg {
		tmpDir, _ := exec.Command("mktemp", "-d").Output()
		// Simplified: in real implementation, use sandbox.VimRunner
		_ = tmpDir
		_ = hintsUsed
		return ChallengeResultMsg{Passed: false, Results: []verify.Result{{Message: "Vim 挑战需要交互式终端"}}}
	}
}

func (m AppModel) launchDockerChallenge(ch *challenge.Challenge, hintsUsed int) tea.Cmd {
	// Use tea.ExecProcess to hand terminal to docker exec
	args := []string{"docker", "exec", "-it", "linuxlab-sandbox", "/bin/bash"}
	c := exec.Command(args[0], args[1:]...)
	return tea.ExecProcess(c, func(err error) tea.Msg {
		// After user exits the shell, run verification
		results := verify.RunAll(ch.Verify)
		passed := verify.AllPassed(results)
		return ChallengeResultMsg{
			Passed:  passed,
			Results: results,
			HintsUsed: hintsUsed,
		}
	})
}
```

```go
// cmd/linuxlab/main.go
package main

import (
	"fmt"
	"os"
	"path/filepath"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/sd3/linuxlab/internal/challenge"
	"github.com/sd3/linuxlab/internal/progress"
	"github.com/sd3/linuxlab/internal/tui"
)

func main() {
	// Determine challenges directory
	challengesDir := "challenges"
	if envDir := os.Getenv("LINUXLAB_CHALLENGES"); envDir != "" {
		challengesDir = envDir
	}

	// Load challenges
	byCategory, err := challenge.LoadAllByCategory(challengesDir)
	if err != nil {
		fmt.Fprintf(os.Stderr, "加载题库失败: %v\n", err)
		fmt.Fprintf(os.Stderr, "请确保 %s 目录存在且包含题目\n", challengesDir)
		os.Exit(1)
	}

	if len(byCategory) == 0 {
		fmt.Fprintf(os.Stderr, "题库为空，请在 %s 目录下添加题目\n", challengesDir)
		os.Exit(1)
	}

	// Load progress
	homeDir, _ := os.UserHomeDir()
	progressDir := filepath.Join(homeDir, ".linuxlab")
	os.MkdirAll(progressDir, 0o755)
	progressPath := filepath.Join(progressDir, "progress.json")

	store, err := progress.NewStore(progressPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "加载进度失败: %v\n", err)
		os.Exit(1)
	}

	// Start TUI
	app := tui.NewAppModel(byCategory, store)
	p := tea.NewProgram(app, tea.WithAltScreen())
	if _, err := p.Run(); err != nil {
		fmt.Fprintf(os.Stderr, "运行失败: %v\n", err)
		os.Exit(1)
	}
}
```

```go
// e2e_test.go
package main

import (
	"os"

func TestE2E_LoadChallengesAndCreateApp(t *testing.T) {
	// Verify the real challenges directory loads correctly
	byCategory, err := challenge.LoadAllByCategory("challenges")
	if err != nil {
		t.Fatalf("LoadAllByCategory failed: %v", err)
	}

	if len(byCategory) < 2 {
		t.Errorf("expected at least 2 categories, got %d", len(byCategory))
	}

	linuxChallenges := byCategory["linux-basics"]
	if len(linuxChallenges) < 5 {
		t.Errorf("linux-basics has %d challenges, want >= 5", len(linuxChallenges))
	}

	vimChallenges := byCategory["vim"]
	if len(vimChallenges) < 5 {
		t.Errorf("vim has %d challenges, want >= 5", len(vimChallenges))
	}

	// Verify each challenge has required fields
	for cat, challenges := range byCategory {
		for _, ch := range challenges {
			if ch.ID == "" {
				t.Errorf("[%s] challenge has empty ID", cat)
			}
			if ch.Title == "" {
				t.Errorf("[%s/%s] has empty Title", cat, ch.ID)
			}
			if ch.Difficulty < 1 || ch.Difficulty > 5 {
				t.Errorf("[%s/%s] difficulty = %d, want 1-5", cat, ch.ID, ch.Difficulty)
			}
			if len(ch.Verify) == 0 {
				t.Errorf("[%s/%s] has no verify rules", cat, ch.ID)
			}
			if ch.Description == "" {
				t.Errorf("[%s/%s] has empty description", cat, ch.ID)
			}
		}
	}
}

func TestE2E_AllSolutionScriptsExist(t *testing.T) {
	// Walk challenges directory and verify each has a solution.sh
	err := filepath.Walk("challenges", func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return err
		}
		if info.Name() == "challenge.yaml" {
			dir := filepath.Dir(path)
			solutionPath := filepath.Join(dir, "solution.sh")
			if _, err := os.Stat(solutionPath); os.IsNotExist(err) {
				t.Errorf("missing solution.sh in %s", dir)
			}
			checkPath := filepath.Join(dir, "check.sh")
			if _, err := os.Stat(checkPath); os.IsNotExist(err) {
				t.Errorf("missing check.sh in %s", dir)
			}
		}
		return nil
	})
	if err != nil {
		t.Fatal(err)
	}
```
